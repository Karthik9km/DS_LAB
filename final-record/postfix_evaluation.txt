PROGRAM POSTFIX_EVALUATION
DEFINE SIZE 20
DECLARE ARRAY stack[SIZE]
SET top = -1

PROCEDURE PUSH ACCEPTS element
    SET top = top + 1
    SET stack[top] = element
END PROCEDURE

PROCEDURE POP
    SET top = top - 1
    RETURN stack[top + 1]
END PROCEDURE

PROCEDURE ISOPERATOR ACCEPTS  element
    IF element='+' OR '-' OR '*' OR '/' OR '^'
        RETURN TRUE
    END IF
END PROCEDURE
